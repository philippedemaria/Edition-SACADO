\begin{ThT}{Les racines carrées\index{Racines carrées}}
Pour tous nombres $a$ et $c$, et $b$ non nul, 
\begin{description}
\item $\frac{a}{b}=\frac{k \times a}{k \times b} , k \neq 0$
\item $\frac{a}{b} + \frac{c}{b}=\frac{a+c}{b}$
\end{description}
\end{ThT}
